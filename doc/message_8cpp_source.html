87 void Micropolis::sendMessages()
88 {
89  short PowerPop;
90  float TM;
91 
92  // Running a scenario, and waiting it to 'end' so we can give a score
93  if (scenario > SC_NONE && scoreType > SC_NONE && scoreWait > 0) {
94  scoreWait--;
95  if (scoreWait == 0) {
96  doScenarioScore(scoreType);
97  }
98  }
99 
100  checkGrowth();
101 
102  totalZonePop = resZonePop + comZonePop + indZonePop;
103  PowerPop = nuclearPowerPop + coalPowerPop;
104 
105  switch (cityTime & 63) {
106 
107  case 1:
108  if (totalZonePop / 4 >= resZonePop) {
109  sendMessage(MESSAGE_NEED_MORE_RESIDENTIAL);
110  }
111  break;
112 
113  case 5:
114  if (totalZonePop / 8 >= comZonePop) {
115  sendMessage(MESSAGE_NEED_MORE_COMMERCIAL);
116  }
117  break;
118 
119  case 10:
120  if (totalZonePop / 8 >= indZonePop) {
121  sendMessage(MESSAGE_NEED_MORE_INDUSTRIAL);
122  }
123  break;
124 
125  case 14:
126  if (totalZonePop > 10 && totalZonePop * 2 > roadTotal) {
127  sendMessage(MESSAGE_NEED_MORE_ROADS);
128  }
129  break;
130 
131  case 18:
132  if (totalZonePop > 50 && totalZonePop > railTotal) {
133  sendMessage(MESSAGE_NEED_MORE_RAILS);
134  }
135  break;
136 
137  case 22:
138  if (totalZonePop > 10 && PowerPop == 0) {
139  sendMessage(MESSAGE_NEED_ELECTRICITY);
140  }
141  break;
142 
143  case 26:
144  if (resPop > 500 && stadiumPop == 0) {
145  sendMessage(MESSAGE_NEED_STADIUM);
146  resCap = true;
147  } else {
148  resCap = false;
149  }
150  break;
151 
152  case 28:
153  if (indPop > 70 && seaportPop == 0) {
154  sendMessage(MESSAGE_NEED_SEAPORT);
155  indCap = true;
156  } else {
157  indCap = false;
158  }
159  break;
160 
161  case 30:
162  if (comPop > 100 && airportPop == 0) {
163  sendMessage(MESSAGE_NEED_AIRPORT);
164  comCap = true;
165  } else {
166  comCap = false;
167  }
168  break;
169 
170  case 32:
171  TM = (float)(unpoweredZoneCount + poweredZoneCount); /* dec score for unpowered zones */
172  if (TM > 0) {
173  if (poweredZoneCount / TM < 0.7) {
174  sendMessage(MESSAGE_BLACKOUTS_REPORTED);
175  }
176  }
177  break;
178 
179  case 35:
180  if (pollutionAverage > /* 80 */ 60) {
181  sendMessage(MESSAGE_HIGH_POLLUTION, -1, -1, true);
182  }
183  break;
184 
185  case 42:
186  if (crimeAverage > 100) {
187  sendMessage(MESSAGE_HIGH_CRIME, -1, -1, true);
188  }
189  break;
190 
191  case 45:
192  if (totalPop > 60 && fireStationPop == 0) {
193  sendMessage(MESSAGE_NEED_FIRE_STATION);
194  }
195  break;
196 
197  case 48:
198  if (totalPop > 60 && policeStationPop == 0) {
199  sendMessage(MESSAGE_NEED_POLICE_STATION);
200  }
201  break;
202 
203  case 51:
204  if (cityTax > 12) {
205  sendMessage(MESSAGE_TAX_TOO_HIGH);
206  }
207  break;
208 
209  case 54:
210  // If roadEffect < 5/8 of max effect
211  if (roadEffect < (5 * MAX_ROAD_EFFECT / 8) && roadTotal > 30) {
212  sendMessage(MESSAGE_ROAD_NEEDS_FUNDING);
213  }
214  break;
215 
216  case 57:
217  // If fireEffect < 0.7 of max effect
218  if (fireEffect < (7 * MAX_FIRE_STATION_EFFECT / 10) && totalPop > 20) {
219  sendMessage(MESSAGE_FIRE_STATION_NEEDS_FUNDING);
220  }
221  break;
222 
223  case 60:
224  // If policeEffect < 0.7 of max effect
225  if (policeEffect < (7 * MAX_POLICE_STATION_EFFECT / 10)
226  && totalPop > 20) {
227  sendMessage(MESSAGE_POLICE_NEEDS_FUNDING);
228  }
229  break;
230 
231  case 63:
232  if (trafficAverage > 60) {
233  sendMessage(MESSAGE_TRAFFIC_JAMS, -1, -1, true);
234  }
235  break;
236 
237  }
238 }

250 void Micropolis::checkGrowth()
251 {
252  if ((cityTime & 3) == 0) {
253  short category = 0;
254  Quad thisCityPop = getPopulation();
255 
256  if (cityPopLast > 0) {
257 
258  CityClass lastClass = getCityClass(cityPopLast);
259  CityClass newClass = getCityClass(thisCityPop);
260 
261  if (lastClass != newClass) {
262 
263  // Switched class, find appropiate message.
264  switch (newClass) {
265 
266  case CC_VILLAGE:
267  // Don't mention it.
268  break;
269 
270  case CC_TOWN:
271  category = MESSAGE_REACHED_TOWN;
272  break;
273 
274  case CC_CITY:
275  category = MESSAGE_REACHED_CITY;
276  break;
277 
278  case CC_CAPITAL:
279  category = MESSAGE_REACHED_CAPITAL;
280  break;
281 
282  case CC_METROPOLIS:
283  category = MESSAGE_REACHED_METROPOLIS;
284  break;
285 
286  case CC_MEGALOPOLIS:
287  category = MESSAGE_REACHED_MEGALOPOLIS;
288  break;
289 
290  default:
291  NOT_REACHED();
292  break;
293 
294  }
295  }
296  }
297 
298  if (category > 0 && category != categoryLast) {
299  sendMessage(category, NOWHERE, NOWHERE, true);
300  categoryLast = category;
301  }
302 
303  cityPopLast = thisCityPop;
304  }
305 }

313 void Micropolis::doScenarioScore(Scenario type)
314 {
315  short z = MESSAGE_SCENARIO_LOST; /* you lose */
316 
317  switch (type) {
318 
319  case SC_DULLSVILLE:
320  if (cityClass >= CC_METROPOLIS) {
321  z = MESSAGE_SCENARIO_WON;
322  }
323  break;
324 
325  case SC_SAN_FRANCISCO:
326  if (cityClass >= CC_METROPOLIS) {
327  z = MESSAGE_SCENARIO_WON;
328  }
329  break;
330 
331  case SC_HAMBURG:
332  if (cityClass >= CC_METROPOLIS) {
333  z = MESSAGE_SCENARIO_WON;
334  }
335  break;
336 
337  case SC_BERN:
338  if (trafficAverage < 80) {
339  z = MESSAGE_SCENARIO_WON;
340  }
341  break;
342 
343  case SC_TOKYO:
344  if (cityScore > 500) {
345  z = MESSAGE_SCENARIO_WON;
346  }
347  break;
348 
349  case SC_DETROIT:
350  if (crimeAverage < 60) {
351  z = MESSAGE_SCENARIO_WON;
352  }
353  break;
354 
355  case SC_BOSTON:
356  if (cityScore > 500) {
357  z = MESSAGE_SCENARIO_WON;
358  }
359  break;
360 
361  case SC_RIO:
362  if (cityScore > 500) {
363  z = MESSAGE_SCENARIO_WON;
364  }
365  break;
366 
367  default:
368  NOT_REACHED();
369  break;
370 
371  }
372 
373  sendMessage(z, NOWHERE, NOWHERE, true, true);
374 
375  if (z == MESSAGE_SCENARIO_LOST) {
376  doLoseGame();
377  }
378 }

402 void Micropolis::doMakeSound(int mesgNum, int x, int y)
403 {
404  assert(mesgNum >= 0);
405 
406  switch (mesgNum) {
407 
408  case MESSAGE_TRAFFIC_JAMS:
409  if (getRandom(5) == 1) {
410  makeSound("city", "HonkHonkMed", x, y);
411  } else if (getRandom(5) == 1) {
412  makeSound("city", "HonkHonkLow", x, y);
413  } else if (getRandom(5) == 1) {
414  makeSound("city", "HonkHonkHigh", x, y);
415  }
416  break;
417 
418  case MESSAGE_HIGH_CRIME:
419  case MESSAGE_FIRE_REPORTED:
420  case MESSAGE_TORNADO_SIGHTED:
421  case MESSAGE_EARTHQUAKE:
422  case MESSAGE_PLANE_CRASHED:
423  case MESSAGE_SHIP_CRASHED:
424  case MESSAGE_TRAIN_CRASHED:
425  case MESSAGE_HELICOPTER_CRASHED:
426  makeSound("city", "Siren", x, y);
427  break;
428 
429  case MESSAGE_MONSTER_SIGHTED:
430  makeSound("city", "Monster", x, y);
431  break;
432 
433  case MESSAGE_FIREBOMBING:
434  makeSound("city", "ExplosionLow", x, y);
435  makeSound("city", "Siren", x, y);
436  break;
437 
438  case MESSAGE_NUCLEAR_MELTDOWN:
439  makeSound("city", "ExplosionHigh", x, y);
440  makeSound("city", "ExplosionLow", x, y);
441  makeSound("city", "Siren", x, y);
442  break;
443 
444  case MESSAGE_RIOTS_REPORTED:
445  makeSound("city", "Siren", x, y);
446  break;
447 
448  }
