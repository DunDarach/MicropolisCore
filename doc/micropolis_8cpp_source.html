804 void Micropolis::simHeat()
805 {
806  int x, y;
807  static int a = 0;
808  short *src, *dst;
809  int fl = heatFlow;
810 
811  const int SRCCOL = WORLD_H + 2;
812  const int DSTCOL = WORLD_H;
813 
814 
815  if (cellSrc == NULL) {
816  cellSrc = (short *)newPtr((WORLD_W + 2) * (WORLD_H + 2) * sizeof (short));
817  cellDst = (short *)&map[0][0];
818  }
819 
820  src = cellSrc + SRCCOL + 1;
821  dst = cellDst;
822 
823  /*
824  * Copy wrapping edges:
825  *
826  * 0 ff f0 f1 ... fe ff f0
827  *
828  * 1 0f 00 01 ... 0e 0f 00
829  * 2 1f 10 11 ... 1e 1f 10
830  * .. .. .. .. .. ..
831  * ef e0 e1 ... ee ef e0
832  * h ff f0 f1 ... fe ff f0
833  *
834  * h+1 0f 00 01 ... 0e 0f 00
835  *
836  * wrap value: effect:
837  * 0 no effect
838  * 1 copy future=>past, no wrap
839  * 2 no copy, wrap edges
840  * 3 copy future=>past, wrap edges
841  * 4 copy future=>past, same edges
842  */
843 
844  switch (heatWrap) {
845  case 0:
846  break;
847  case 1:
848  for (x = 0; x < WORLD_W; x++) {
849  memcpy(src, dst, WORLD_H * sizeof (short));
850  src += SRCCOL;
851  dst += DSTCOL;
852  }
853  break;
854  case 2:
855  for (x = 0; x < WORLD_W; x++) {
856  src[-1] = src[WORLD_H - 1];
857  src[WORLD_H] = src[0];
858  src += SRCCOL;
859  dst += DSTCOL;
860  }
861  memcpy(
862  cellSrc,
863  cellSrc + (SRCCOL * WORLD_W),
864  SRCCOL * sizeof (short));
865  memcpy(
866  cellSrc + SRCCOL * (WORLD_W + 1),
867  cellSrc + SRCCOL,
868  SRCCOL * sizeof (short));
869  break;
870  case 3:
871  for (x = 0; x < WORLD_W; x++) {
872  memcpy(src, dst, WORLD_H * sizeof (short));
873  src[-1] = src[WORLD_H - 1];
874  src[WORLD_H] = src[0];
875  src += SRCCOL;
876  dst += DSTCOL;
877  }
878  memcpy(
879  cellSrc,
880  cellSrc + (SRCCOL * WORLD_W),
881  SRCCOL * sizeof (short));
882  memcpy(
883  cellSrc + SRCCOL * (WORLD_W + 1),
884  cellSrc + SRCCOL,
885  SRCCOL * sizeof (short));
886  break;
887  case 4:
888  src[0] =
889  dst[0];
890  src[1 + WORLD_H] =
891  dst[WORLD_H - 1];
892  src[(1 + WORLD_W) * SRCCOL] =
893  dst[(WORLD_W - 1) * DSTCOL];
894  src[((2 + WORLD_W) * SRCCOL) - 1] =
895  dst[(WORLD_W * WORLD_H) - 1];
896  for (x = 0; x < WORLD_W; x++) {
897  memcpy(src, dst, WORLD_H * sizeof (short));
898  src[-1] = src[0];
899  src[WORLD_H] = src[WORLD_H - 1];
900  src += SRCCOL;
901  dst += DSTCOL;
902  }
903  memcpy(
904  cellSrc + (SRCCOL * (WORLD_W + 1)),
905  cellSrc + (SRCCOL * WORLD_W),
906  SRCCOL * sizeof (short));
907  memcpy(
908  cellSrc,
909  cellSrc + SRCCOL,
910  SRCCOL * sizeof (short));
911  break;
912  default:
913  NOT_REACHED();
914  break;
915  }
916 
917 
918 #define CLIPPER_LOOP_BODY(CODE) \
919  src = cellSrc; dst = cellDst; \
920  for (x = 0; x < WORLD_W;) { \
921  short nw, n, ne, w, c, e, sw, s, se; \
922  src = cellSrc + (x * SRCCOL); \
923  dst = cellDst + (x * DSTCOL); \
924  w = src[0]; c = src[SRCCOL]; e = src[2 * SRCCOL]; \
925  sw = src[1]; s = src[SRCCOL + 1]; se = src[(2 * SRCCOL) + 1]; \
926  for (y = 0; y < WORLD_H; y++) { \
927  nw = w; w = sw; sw = src[2]; \
928  n = c; c = s; s = src[SRCCOL + 2]; \
929  ne = e; e = se; se = src[(2 * SRCCOL) + 2]; \
930  { CODE } \
931  src++; dst++; \
932  } \
933  x++; \
934  src = cellSrc + ((x + 1) * SRCCOL) - 3; \
935  dst = cellDst + ((x + 1) * DSTCOL) - 1; \
936  nw = src[1]; n = src[SRCCOL + 1]; ne = src[(2 * SRCCOL) + 1]; \
