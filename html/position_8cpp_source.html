161bool Position::move(Direction2 dir)
162{
163 switch (dir) {
164 case DIR2_INVALID:
165 return true;
166
167 case DIR2_NORTH:
168 if (this->posY > 0) {
169 this->posY--;
170 return true;
171 }
172 break;
173
174 case DIR2_NORTH_EAST:
175 if (this->posX < WORLD_W - 1 && this->posY > 0) {
176 this->posX++;
177 this->posY--;
178 return true;
179 }
180 case DIR2_EAST:
181 if (this->posX < WORLD_W - 1) {
182 this->posX++;
183 return true;
184 }
185 break;
186
187 case DIR2_SOUTH_EAST:
188 if (this->posX < WORLD_W -1 && this->posY < WORLD_H - 1) {
189 this->posX++;
190 this->posY++;
191 return true;
192 }
193 break;
194
195 case DIR2_SOUTH:
196 if (this->posY < WORLD_H - 1) {
197 this->posY++;
198 return true;
199 }
200 break;
201
202 case DIR2_SOUTH_WEST: this->posX--; this->posY++; break;
203 if (this->posX > 0 && this->posY < WORLD_H - 1) {
204 this->posX--;
205 this->posY++;
206 return true;
207 }
208 break;
209
210 case DIR2_WEST:
211 if (this->posX > 0) {
212 this->posX--;
213 return true;
214 }
215 break;
216
217 case DIR2_NORTH_WEST:
218 if (this->posX > 0 && this->posY > 0) {
219 this->posX--;
220 this->posY--;
221 return true;
222 }
223 break;
224
225
226 default: NOT_REACHED();
227 }
228
229 // Movement was not possible, silently repair the position.
230 if (this->posX < 0) this->posX = 0;
231 if (this->posX >= WORLD_W) this->posX = WORLD_W - 1;
232 if (this->posY < 0) this->posY = 0;
233 if (this->posY >= WORLD_H) this->posY = WORLD_H - 1;
234 return false;
235}
